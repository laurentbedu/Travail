-- Update le type utilisateur d'inscrit à visiteur
UPDATE TypeUtilisateur SET libelletype = "Visiteur" WHERE libelletype = "Inscrit";

-- Suppression dans Statut de "En attente de validation"
DELETE FROM Statut WHERE libelleStatut = "En attente de validation";

-- Selectionner tout dans TypeMedia
SELECT * from TypeMedia;

-- Le nombre de Statut dans la table Statut
SELECT COUNT(*) FROM Statut;

-- Création d'un alias
SELECT COUNT(*) AS "Nombre de statuts" FROM Statut;


-- Liste des utilisateurs par ordre alphabétique
SELECT * FROM Utilisateur ORDER BY login ASC;

-- Liste des utilisateurs par date de naissance du plus jeune au plus vieux
SELECT * FROM Utilisateur ORDER BY dateNaissance DESC;

-- Liste des administrateurs
SELECT * FROM Utilisateur WHERE typeUtilisateur=3;

-- liste des autres utilisateurs
SELECT * FROM Utilisateur WHERE typeUtilisateur<3;

-- Pseudo et mot de passe des utilisateurs
SELECT login AS "Pseudo", motDepasse AS "Mot de passe" FROM Utilisateur WHERE typeUtilisateur = 3;

-- Requête imbriquée
SELECT login AS "Pseudo", motDepasse AS "Mot de passe" FROM Utilisateur WHERE typeUtilisateur =
(SELECT typeUtilisateur FROM typeUtilisateur WHERE libelleType="Administrateur"); 

-- Jointure avec ALIAS
SELECT U.login AS "Pseudo", U.motDePasse AS "mdp", T.libelleType AS "Statut" FROM Utilisateur AS U INNER JOIN typeUtilisateur AS T ON U.typeUtilisateur = T.idtypeUtilisateur;

-- Grouper
SELECT U.login AS "Pseudo", U.motDePasse AS "mdp", T.libelleType AS "Statut" FROM Utilisateur AS U INNER JOIN typeUtilisateur AS T ON U.typeUtilisateur = T.idtypeUtilisateur GROUP BY T.libelleType;

-- Sélectionner le dernier article publié
SELECT * FROM Article WHERE statut IN(SELECT idStatut FROM Statut WHERE libelleStatut = "Publié") ORDER BY dateConception DESC LIMIT 0, 1;

-- Tous les articles qui commencent par "U"
SELECT * FROM Article WHERE Titre LIKE "U%";

-- Remplacer la lettre "U" de chaque Titre par un "A"
UPDATE Article SET Titre = REPLACE(Titre, "U", "A");

-- Suppression les utilisateurs qui n'ont pas publié d'article et de commentaire
DELETE FROM Utilisateur WHERE idUtilisateur NOT IN(SELECT auteur FROM Article) AND idUtilisateur NOT IN(SELECT auteur FROM Commentaire);
